Database Setup — run once in Supabase SQL Editor

⚠ If saves are not working, RLS (Row Level Security) is blocking writes. Run the SQL below to fix everything at once.

Go to supabase.com → your project → SQL Editor → New query, paste everything below, click Run.

-- ① Create all tables
CREATE TABLE IF NOT EXISTS properties (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name text, slug text UNIQUE, location text, area text,
  bhk text, units int, rent int, status text DEFAULT 'available',
  description text, map_embed text, sort_order int DEFAULT 1,
  amenities jsonb DEFAULT '[]', images jsonb DEFAULT '[]',
  created_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS enquiries (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text, phone text, email text,
  property_slug text, property_interest text,
  move_in_date text, budget text, message text,
  intent text DEFAULT 'tenant', contacted boolean DEFAULT false,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS grid_members (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  role text, time text, quote text, img text, sort_order int DEFAULT 1
);
CREATE TABLE IF NOT EXISTS grid_requests (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  first_name text, last_name text, email text,
  what_they_do text, preferred_property text,
  moving_when text, message text,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS site_content (
  id int PRIMARY KEY DEFAULT 1, data jsonb
);
INSERT INTO site_content (id, data) VALUES (1, '{}') ON CONFLICT (id) DO NOTHING;

-- ② Disable RLS so admin saves work (fixes the "save not working" issue)
ALTER TABLE properties DISABLE ROW LEVEL SECURITY;
ALTER TABLE enquiries DISABLE ROW LEVEL SECURITY;
ALTER TABLE grid_members DISABLE ROW LEVEL SECURITY;
ALTER TABLE grid_requests DISABLE ROW LEVEL SECURITY;
ALTER TABLE site_content DISABLE ROW LEVEL SECURITY;
